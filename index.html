C# — строго типизированный язык программирования, в котором каждый объект и переменная имеет определённый тип данных.

1.2. Операторы

  • Арифметические: +, -, *, /, %
  • Сравнения: ==, !=, <, >, <=, >=
  • Логические: &&, ||, !
  • Присваивания: =, +=, -=, *=, /=
  • Тернарный оператор: условие ? значение1 : значение2

1.3. Ветвления

int temperature = 10;

if (temperature > 25)
    Console.WriteLine("Жарко");
else if (temperature > 10)
    Console.WriteLine("Тепло");
else
    Console.WriteLine("Холодно");

1.4. Циклы

// for
for (int i = 0; i < 5; i++)
    Console.WriteLine(i);

// while
int n = 0;
while (n < 3)
{
    Console.WriteLine("Повтор " + n);
    n++;
}

// foreach
string[] names = { "Иван", "Анна", "Пётр" };
foreach (string n in names)
    Console.WriteLine(n);

2. Принципы объектно-ориентированного программирования (ООП)

ООП основано на идее моделирования реальных сущностей с помощью классов и объектов.

2.1. Класс и объект

class Student
{
    public string Name;
    public int Age;

    public void SayHello()
    {
        Console.WriteLine($"Привет! Меня зовут {Name}, мне {Age} лет.");
    }
}

Student s = new Student();
s.Name = "Анна";
s.Age = 19;
s.SayHello();

2.2. Инкапсуляция

Сокрытие данных внутри класса с помощью private и открытый доступ через public методы.

class Account
{
    private double balance;

    public void Deposit(double amount)
    {
        balance += amount;
    }

    public double GetBalance() => balance;
}

2.3. Наследование

class Person
{
    public string Name;
    public void Introduce() => Console.WriteLine($"Я — {Name}");
}

class Employee : Person
{
    public string Position;
}

2.4. Полиморфизм

class Animal { public virtual void Speak() => Console.WriteLine("Животное говорит"); }
class Dog : Animal { public override void Speak() => Console.WriteLine("Гав!"); }

Animal a = new Dog();
a.Speak(); // Гав!

  +Position : string
}

@enduml

3. Работа с коллекциями и обработка исключений

3.1. Коллекции

using System.Collections.Generic;

List<string> names = new() { "Иван", "Анна", "Пётр" };
names.Add("Мария");
foreach (var n in names) Console.WriteLine(n);

Dictionary<string, int> grades = new()
{
    ["Математика"] = 5,
    ["Информатика"] = 4
};
Console.WriteLine(grades["Информатика"]);

3.2. Обработка исключений

try
{
    int a = 10, b = 0;
    Console.WriteLine(a / b);
}
catch (DivideByZeroException ex)
{
    Console.WriteLine("Ошибка деления: " + ex.Message);
}
finally
{
    Console.WriteLine("Программа завершена.");
}

6. Принципы интеграции и взаимодействия модулей

Интеграция — это процесс соединения независимых частей программы в единую систему.

class Logger
{
    public void Log(string message) => Console.WriteLine("[LOG] " + message);
}

class Processor
{
    private Logger logger = new();

    public void Run()
    {
        logger.Log("Процесс запущен");
    }
}

PlantUML — взаимодействие модулей

@startuml
participant Program
participant Logger
participant Processor

Program -> Processor : вызов Run()
Processor -> Logger : Log("Процесс запущен")
Logger --> Processor : сообщение выведено
Processor --> Program : завершение
@enduml
